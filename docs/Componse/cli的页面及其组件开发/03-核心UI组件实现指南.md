```typescript
import { Box, Text, useInput } from 'ink'
import { useState, useMemo } from 'react'
import fuzzysort from 'fuzzysort'
import { TextInput } from 'ink-text-input'

export interface DialogSelectOption<T = any> {
  label: string             // æ˜¾ç¤ºæ–‡æœ¬
  value: T                  // å€¼
  description?: string      // æè¿°
  category?: string         // åˆ†ç±»
  disabled?: boolean        // ç¦ç”¨
  footer?: string           // åº•éƒ¨ä¿¡æ¯
}

export interface DialogSelectProps<T> {
  title: string
  placeholder?: string
  options: DialogSelectOption<T>[]
  current?: T               // å½“å‰é€‰ä¸­çš„å€¼
  onSelect?: (option: DialogSelectOption<T>) => void
  onClose?: () => void
}

export function DialogSelect<T>({
  title,
  placeholder = 'Search...',
  options,
  current,
  onSelect,
  onClose,
}: DialogSelectProps<T>) {
  const [query, setQuery] = useState('')
  const [selectedIndex, setSelectedIndex] = useState(0)

  // æ¨¡ç³Šæœç´¢
  const filtered = useMemo(() => {
    if (!query) return options.filter(o => !o.disabled)

    const results = fuzzysort.go(query, options, {
      keys: ['label', 'description', 'category'],
      threshold: -10000,
    })

    return results.map(r => r.obj).filter(o => !o.disabled)
  }, [query, options])

  // åˆ†ç»„
  const grouped = useMemo(() => {
    const map = new Map<string, DialogSelectOption<T>[]>()

    filtered.forEach(option => {
      const category = option.category || ''
      if (!map.has(category)) {
        map.set(category, [])
      }
      map.get(category)!.push(option)
    })

    return Array.from(map.entries())
  }, [filtered])

  // æ‰å¹³åŒ–ï¼ˆç”¨äºç´¢å¼•ï¼‰
  const flatOptions = useMemo(() =>
    grouped.flatMap(([_, opts]) => opts),
    [grouped]
  )

  // é”®ç›˜å¯¼èˆª
  useInput((input, key) => {
    if (key.upArrow) {
      setSelectedIndex(prev =>
        prev === 0 ? flatOptions.length - 1 : prev - 1
      )
    }

    if (key.downArrow) {
      setSelectedIndex(prev =>
        prev === flatOptions.length - 1 ? 0 : prev + 1
      )
    }

    if (key.return) {
      const selected = flatOptions[selectedIndex]
      if (selected) {
        onSelect?.(selected)
      }
    }

    if (key.escape) {
      onClose?.()
    }
  })

  const selectedOption = flatOptions[selectedIndex]

  return (
    <Box flexDirection="column" padding={1}>
      {/* æ ‡é¢˜æ  */}
      <Box justifyContent="space-between" marginBottom={1}>
        <Text bold color="cyan">{title}</Text>
        <Text dimColor>esc</Text>
      </Box>

      {/* æœç´¢æ¡† */}
      <Box marginBottom={1}>
        <Text dimColor>Search: </Text>
        <TextInput
          value={query}
          onChange={setQuery}
          placeholder={placeholder}
        />
      </Box>

      {/* é€‰é¡¹åˆ—è¡¨ */}
      <Box flexDirection="column" gap={1}>
        {grouped.map(([category, opts], groupIndex) => (
          <Box key={category || groupIndex} flexDirection="column">
            {/* åˆ†ç±»æ ‡é¢˜ */}
            {category && (
              <Box marginBottom={0.5}>
                <Text bold color="yellow">{category}</Text>
              </Box>
            )}

            {/* é€‰é¡¹ */}
            {opts.map((option, index) => {
              const globalIndex = flatOptions.indexOf(option)
              const isSelected = globalIndex === selectedIndex
              const isCurrent = option.value === current

              return (
                <Box
                  key={index}
                  paddingX={1}
                  backgroundColor={isSelected ? 'blue' : undefined}
                >
                  {/* å½“å‰æ ‡è®° */}
                  {isCurrent && (
                    <Text color={isSelected ? 'white' : 'blue'}>â— </Text>
                  )}

                  {/* æ ‡ç­¾ */}
                  <Box flexGrow={1}>
                    <Text
                      bold={isSelected}
                      color={isSelected ? 'white' : 'green'}
                    >
                      {option.label}
                    </Text>

                    {/* æè¿° */}
                    {option.description && (
                      <Text dimColor> - {option.description}</Text>
                    )}
                  </Box>

                  {/* åº•éƒ¨ä¿¡æ¯ */}
                  {option.footer && (
                    <Text dimColor>{option.footer}</Text>
                  )}
                </Box>
              )
            })}
          </Box>
        ))}
      </Box>

      {/* åº•éƒ¨æç¤º */}
      {selectedOption && (
        <Box marginTop={1} borderStyle="single" borderColor="gray" padding={1}>
          <Text dimColor>{selectedOption.description || 'No description'}</Text>
        </Box>
      )}
    </Box>
  )
}
```

```typescript
import { useState, useEffect } from 'react'
import { Storage } from '@/storage'

export function useHistory() {
  const [history, setHistory] = useState<string[]>([])

  // åŠ è½½å†å²
  useEffect(() => {
    Storage.loadHistory().then(setHistory)
  }, [])

  // æ·»åŠ åˆ°å†å²
  const addToHistory = (text: string) => {
    const newHistory = [...history, text].slice(-50)  // ä¿ç•™æœ€è¿‘ 50 æ¡
    setHistory(newHistory)
    Storage.saveHistory(newHistory)
  }

  // å¯¼èˆªå†å²
  const navigate = (direction: 'up' | 'down', currentIndex: number) => {
    if (direction === 'up' && currentIndex < history.length - 1) {
      return currentIndex + 1
    }
    if (direction === 'down' && currentIndex > -1) {
      return currentIndex - 1
    }
    return currentIndex
  }

  return { history, addToHistory, navigate }
}
```

```typescript
// src/contexts/dialog.tsx
import { createContext, useContext, useState, ReactNode } from 'react'

interface DialogContextValue {
  stack: ReactNode[]
  push: (dialog: ReactNode) => void
  pop: () => void
  replace: (dialog: ReactNode) => void
  clear: () => void
}

const DialogContext = createContext<DialogContextValue | null>(null)

export function DialogProvider({ children }: { children: ReactNode }) {
  const [stack, setStack] = useState<ReactNode[]>([])

  const push = (dialog: ReactNode) => {
    setStack(prev => [...prev, dialog])
  }

  const pop = () => {
    setStack(prev => prev.slice(0, -1))
  }

  const replace = (dialog: ReactNode) => {
    setStack([dialog])
  }

  const clear = () => {
    setStack([])
  }

  return (
    <DialogContext.Provider value={{ stack, push, pop, replace, clear }}>
      {children}
    </DialogContext.Provider>
  )
}

export function useDialog() {
  const context = useContext(DialogContext)
  if (!context) {
    throw new Error('useDialog must be used within DialogProvider')
  }
  return context
}
```

```typescript
// src/ui/toast.tsx
import { Box, Text } from 'ink'
import { useState, useEffect } from 'react'

export interface ToastProps {
  message: string
  variant?: 'info' | 'success' | 'warning' | 'error'
  duration?: number
  onClose?: () => void
}

export function Toast({ message, variant = 'info', duration = 3000, onClose }: ToastProps) {
  useEffect(() => {
    const timer = setTimeout(() => {
      onClose?.()
    }, duration)

    return () => clearTimeout(timer)
  }, [duration, onClose])

  const colors = {
    info: 'blue',
    success: 'green',
    warning: 'yellow',
    error: 'red',
  }

  return (
    <Box
      borderStyle="round"
      borderColor={colors[variant]}
      paddingX={2}
      paddingY={1}
    >
      <Text color={colors[variant]}>{message}</Text>
    </Box>
  )
}

// Toast Provider
export function ToastProvider({ children }: { children: ReactNode }) {
  const [toasts, setToasts] = useState<ToastProps[]>([])

  const show = (toast: ToastProps) => {
    const id = Date.now()
    setToasts(prev => [...prev, { ...toast, id }])
  }

  const remove = (id: number) => {
    setToasts(prev => prev.filter(t => t.id !== id))
  }

  return (
    <ToastContext.Provider value={{ show }}>
      {children}

      {/* Toast å®¹å™¨ */}
      <Box position="absolute" top={0} right={2} flexDirection="column" gap={1}>
        {toasts.map((toast, index) => (
          <Toast
            key={index}
            {...toast}
            onClose={() => remove(toast.id!)}
          />
        ))}
      </Box>
    </ToastContext.Provider>
  )
}

// Hook
export function useToast() {
  const context = useContext(ToastContext)
  return {
    info: (message: string) => context.show({ message, variant: 'info' }),
    success: (message: string) => context.show({ message, variant: 'success' }),
    warning: (message: string) => context.show({ message, variant: 'warning' }),
    error: (message: string) => context.show({ message, variant: 'error' }),
  }
}
```
